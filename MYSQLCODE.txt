Create the Database: 
CREATE DATABASE studentsManagement;

Use the Database: 
USE studentsManagement;

Create the Students Table:
CREATE TABLE students (
                          student_id INT AUTO_INCREMENT PRIMARY KEY,
                          name VARCHAR(255),
                          age INT,
                          gender ENUM('Male', 'Female', 'Other'),
                          course VARCHAR(255)
                          );
